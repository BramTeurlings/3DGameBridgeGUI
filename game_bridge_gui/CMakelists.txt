cmake_minimum_required(VERSION 3.23)
project (GameBridgeGUI VERSION 0.1)

add_executable(GameBridgeGUI
src/main.cpp
)

set_target_properties(GameBridgeGUI PROPERTIES LINKER_LANGUAGE CXX)

target_link_libraries(GameBridgeGUI PRIVATE GameBridgeApi)

# Copy all linked dlls to build executable directory
add_custom_command(
    TARGET GameBridgeGUI POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different $<TARGET_RUNTIME_DLLS:GameBridgeGUI> $<TARGET_FILE_DIR:GameBridgeGUI>
)
